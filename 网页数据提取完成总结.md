### 5.3 数据合并

```python
import pandas as pd

data_dir = '网页提取数据'
all_files = [f for f in os.listdir(data_dir) if f.endswith('.csv')]

dfs = []
for file in all_files:
    df = pd.read_csv(os.path.join(data_dir, file), encoding='utf-8-sig')
    dfs.append(df)

combined_df = pd.concat(dfs, ignore_index=True)
```
